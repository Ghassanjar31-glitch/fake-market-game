    <script>
        // تهيئة محرك التليجرام المصغر
        const tg = window.Telegram.WebApp;
        tg.expand();

        // قاعدة بيانات الحالة المالية والعملات
        let userStars = 5000;
        let todayProfit = 0;
        let activeAsset = 'CYBER';

        // بيانات المحفظة الخاصة بالمستخدم للأصول المشترية
        let portfolio = { CYBER: 0, AERO: 0, STAR: 0 };

        // بيانات محاكاة السوق للعملات المتعددة
        let marketData = {
            CYBER: { price: 120, history: [], lastChange: 0 },
            AERO: { price: 450, history: [], lastChange: 0 },
            STAR: { price: 85, history: [], lastChange: 0 }
        };

        // توليد حركة بيانات أولية عشوائية لكل عملة لتعبئة الرسم البياني
        const baseTime = new Date().getTime() - 300000;
        Object.keys(marketData).forEach(coin => {
            let initialPrice = marketData[coin].price;
            for(let i=0; i<15; i++) {
                let open = initialPrice + Math.floor(Math.random() * 20) - 10;
                let close = open + Math.floor(Math.random() * 20) - 10;
                let high = Math.max(open, close) + Math.floor(Math.random() * 5);
                let low = Math.min(open, close) - Math.floor(Math.random() * 5);
                marketData[coin].history.push({x: baseTime + (i * 20000), y: [open, high, low, close]});
                initialPrice = close;
            }
            marketData[coin].price = initialPrice;
        });

        // بناء وضبط الرسم البياني التفاعلي
        const chartOptions = {
            series: [{ data: marketData['CYBER'].history }],
            chart: { type: 'candlestick', height: 240, toolbar: { show: false }, animations: { enabled: false } },
            xaxis: { type: 'datetime', labels: { show: false } },
            yaxis: { tooltip: { enabled: false }, labels: { style: { colors: '#6c7d93' } } },
            grid: { borderColor: '#171e2e' }
        };
        const mainChart = new ApexCharts(document.querySelector("#crypto-chart"), chartOptions);
        mainChart.render();

        // المحرك الحي لتحديث الأسواق المتعددة بشكل عشوائي ومستمر (كل ثانية)
        setInterval(() => {
            Object.keys(marketData).forEach(coin => {
                let coinObj = marketData[coin];
                let lastBar = coinObj.history[coinObj.history.length - 1];
                let open = lastBar.y[3];
                let factor = coin === 'AERO' ? 8 : 3; // تقلبات أكبر لعملة ايرو
                let change = Math.floor(Math.random() * (factor * 2)) - factor;
                let close = open + change;
                let high = Math.max(open, close) + Math.floor(Math.random() * 2);
                let low = Math.min(open, close) - Math.floor(Math.random() * 2);

                coinObj.price = close;
                coinObj.lastChange = (change / open * 100);

                // استبدال النقطة الأقدم بالنقطة السعرية الحالية
                coinObj.history.shift();
                coinObj.history.push({x: new Date().getTime(), y: [open, high, low, close]});

                // تحديث الأسعار في شريط الاختيار العلوي
                const tickerEl = document.getElementById(`ticker-${coin}`);
                if (tickerEl) {
                    tickerEl.innerText = `${close} ⭐`;
                    tickerEl.className = change >= 0 ? 'crypto-tab-price up-trend' : 'crypto-tab-price down-trend';
                }
            });

            // تحديث الشاشة الحالية المفتوحة للمستخدم بالبيانات الفورية
            updateCurrentUIValues();
        }, 1200);

        // وظيفة تحديث الشاشة بناءً على التغييرات المالية المستمرة
        function updateCurrentUIValues() {
            let activeObj = marketData[activeAsset];
            
            // تحديث واجهة التداول الرئيسية
            document.getElementById('active-asset-price').innerText = `${activeObj.price} ⭐ (${activeObj.lastChange >= 0 ? '+' : ''}${activeObj.lastChange.toFixed(2)}%)`;
            document.getElementById('active-asset-price').className = activeObj.lastChange >= 0 ? 'up-trend' : 'down-trend';
            
            if (document.getElementById('page-trade').classList.contains('active')) {
                mainChart.updateSeries([{ data: activeObj.history }]);
            }

            // تحديث واجهة المحفظة الاستثمارية بالقيم المالية الفعلية
            Object.keys(portfolio).forEach(coin => {
                document.getElementById(`port-owned-${coin}`).innerText = portfolio[coin].toFixed(2);
                let totalValue = portfolio[coin] * marketData[coin].price;
                document.getElementById(`port-val-${coin}`).innerText = `${Math.floor(totalValue).toLocaleString()} ⭐`;
            });
        }

        // التبديل بين العملات الرقمية المتوفرة
        function switchAsset(coin) {
            activeAsset = coin;
            
            // إزالة الكلاس النشط وتطبيقه على التاب الجديد
            document.querySelectorAll('.crypto-tab').forEach(el => el.classList.remove('active'));
            document.getElementById(`tab-${coin}`).classList.add('active');

            // تغيير العناوين وتحديث الرسم البياني فوراً
            document.getElementById('active-asset-title').innerText = `تداول العملة: ${coin}/STARS`;
            mainChart.updateSeries([{ data: marketData[coin].history }]);
            updateCurrentUIValues();

            if(tg.HapticFeedback) tg.HapticFeedback.impactOccurred('light');
        }

        // معالجة عمليات الشراء والبيع للأصول
        function tradeAction(type) {
            let pricePerUnit = marketData[activeAsset].price;

            if (type === 'BUY') {
                if (userStars >= pricePerUnit) {
                    userStars -= pricePerUnit;
                    portfolio[activeAsset] += 1;
                    if(tg.HapticFeedback) tg.HapticFeedback.impactOccurred('medium');
                } else {
                    alert("عذراً، رصيدك الحالي من النجوم غير كافٍ لإتمام عملية الشراء.");
                }
            } else if (type === 'SELL') {
                if (portfolio[activeAsset] >= 1) {
                    userStars += pricePerUnit;
                    portfolio[activeAsset] -= 1;
                    todayProfit += Math.floor(pricePerUnit * 0.1); // محاكاة أرباح مستقطعة
                    if(tg.HapticFeedback) tg.HapticFeedback.impactOccurred('success');
                } else {
                    alert("لا توجد لديك وحدات كافية من هذه العملة لبيعها.");
                }
            }

            // تحديث الرصيد الإجمالي في جميع اللوحات
            document.getElementById('global-stars').innerText = `${userStars.toLocaleString()} ⭐`;
            document.getElementById('global-profit').innerText = `+${todayProfit.toLocaleString()} ⭐`;
            document.getElementById('user-rank-profit').innerText = `+${todayProfit.toLocaleString()} ⭐`;
            
            updateCurrentUIValues();
        }

        // نظام التنقل بين الواجهات (Tabs Navigation)
        function navigateTo(pageId) {
            // إخفاء كل الواجهات وإلغاء تنشيط أزرار التحكم السفلية
            document.querySelectorAll('.page').forEach(page => page.classList.remove('active'));
            document.querySelectorAll('.nav-item').forEach(item => item.classList.remove('active'));

            // إظهار الواجهة المطلوبة وتفعيل زرها المناسب
            document.getElementById(`page-${pageId}`).classList.add('active');
            document.getElementById(`nav-${pageId}`).classList.add('active');

            // تحديث البيانات فور الانتقال للتأكد من المزامنة
            updateCurrentUIValues();
            if(tg.HapticFeedback) tg.HapticFeedback.impactOccurred('light');
        }
    </script>
